    PROBLEM
    =======

    GENERATE THIS REPORT FOR ANY DATASET DYNAMICALLY
    This was the accepted solution?

    proc report data=sashelp.class out=want;
    cols  NAME  =f1
          SEX   =f2
          AGE   =f3;
    define f1 /display "NAME";
    define f2 /display "SEX";
    define f3 /display "AGE";
    run;

    %deletesasmacn;
    %symdel new / nowarn;
    %arraydelete(_fs);

    %array(_vs
     ,values=%utl_varlist(class));

    %array(_fs,values=1-&_vsn);

    /*---
    GLOBAL _FS1 1
    GLOBAL _FS2 2
    GLOBAL _FS3 3

    GLOBAL _VS1 NAME
    GLOBAL _VS2 SEX
    GLOBAL _VS3 AGE
    GLOBAL _VSN 3
    ---*/

    proc report data=class list out=want ;
     cols
      %do_over(_vs _fs,phrase=%str(?_vs=f?_fs));

     %do_over(_fs _vs,phrase=
      %str(define f?_fs /display "?_vs";))

    run;quit;
